#
# Testing
#

ENABLE_TESTING()

# Automatically create tests using files named "Test*.cpp"
FILE(GLOB TEST_PROGS "*Test*.cpp")
FOREACH(TEST_PROG ${TEST_PROGS})
    GET_FILENAME_COMPONENT(TEST_ROOT ${TEST_PROG} NAME_WE)
    ADD_EXECUTABLE(${TEST_ROOT} ${TEST_PROG})
    TARGET_LINK_LIBRARIES(${TEST_ROOT} ${OpenMM_LIBRARIES} ${NML_PLUGIN} optimized ${OpenMM_SERIALIZATION_LIBRARY} debug ${OpenMM_SERIALIZATION_DEBUG_LIBRARY})
    ADD_TEST(${TEST_ROOT} ${EXECUTABLE_OUTPUT_PATH}/${TEST_ROOT})
ENDFOREACH(TEST_PROG ${TEST_PROGS})

FILE(GLOB TEST_FILES *.txt *.xml)

#
# Copy files to run dir
#
FOREACH (dataFile ${TEST_FILES})
    ADD_CUSTOM_COMMAND(TARGET ${TEST_ROOT}
        COMMAND ${CMAKE_COMMAND} ARGS -E copy ${dataFile} ${CMAKE_BINARY_DIR})
ENDFOREACH (dataFile)
